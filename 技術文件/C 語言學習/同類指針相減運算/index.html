#include <stdio.h>
int main(){
    int arr[5] = {1, 3, 5, 7, 9};

    // 使用 & 運算符，提取 arr[0] 的地址存入指針變數 p1
    int *p1 = &arr[0];
    int *p2 = &arr[3];
    printf("%d\n", *p1); // 1
    printf("%d\n", *p2); // 7
    printf("%d\n", *p2-*p1); // 用 * 運算符取 p2、p1 指針值做相減(7-1=6)
    printf("%d\n", p1); // 指針相減 p2-p1 相差 14
    printf("%d\n", p2); // 指針相減 p2-p1 相差 14
    printf("%d\n", p2 - p1); // 指針相減 p2-p1 相差 12 個 byte (1807002288-1807002300 = 12)/4bit(int 長度) = 3 個元素的距離
    return  0;
}

// execute: clang PointerTest-2.3.3.c -o PointerTest-2.3.3 && ./PointerTest-2.3.3

& 與 * 差異
& -> 從「變數」取址運算符
* -> 從「指針」取值運算符
