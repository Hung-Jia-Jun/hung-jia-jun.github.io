題目:
	輸入班級所有學生的成績，輸出成績高於平均的學生姓名與分數
``` C
#include <string.h>
#include <stdio.h>
#define LENGTH 4
// 需求: 輸入班級學生訊息
struct Student
{
    int id;
    char name[20];
    char gender;
    int score;
};

int main()
{
    // 聲明學生的結構體數組
    struct Student student_arr[LENGTH] = {{.id=1, .name="alice", .gender='M', .score=60},
                                          {.id=2, .name="bob", .gender='F', .score=65},
                                          {.id=3, .name="hoho", .gender='F', .score=70},
                                          {.id=4, .name="doe", .gender='M', .score=80}};
    int score_sum = 0;
    for (int i = 0; i < LENGTH; i++){
        printf("score: %d\n", student_arr[i].score);
        score_sum += student_arr[i].score;
    }
    int avg_score = score_sum/LENGTH;
    printf("avg_score: %d\n", avg_score);
    for (int i = 0; i < LENGTH; i++){
        if (student_arr[i].score > avg_score){
            printf("name: %s, score: %d\n", student_arr[i].name, student_arr[i].score);
        }
    }
}

output: 
	score: 60
	score: 65
	score: 70
	score: 80
	avg_score: 68
	name: hoho, score: 70
	name: doe, score: 80
```
