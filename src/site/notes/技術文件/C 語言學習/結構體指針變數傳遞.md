```C
#include <string.h>
#include <stdio.h>
/*結構體傳參*/
// 聲明一個結構體
struct Person
{
   char name[20];
   int age;
};

void addAge(struct Person per){
    // 實際傳入的是結構體的副本
    per.age = per.age + 1;
}
void addAge1(struct Person *per){
    // 修改的是指針裡面的值
    per->age = per->age + 1;
}

// 結構體指針
int main()
{
    struct Person per1 = {.name="Tom", .age=13};
    // 實際傳入的是結構體的副本，pass by value
    addAge(per1);
    printf("name: %s\n", per1.name);
    printf("age: %d\n", per1.age);

    // 傳入結構體的指針，pass by reference
    addAge1(&per1);
    printf("name: %s\n", per1.name);
    printf("age: %d\n", per1.age);
}
```
