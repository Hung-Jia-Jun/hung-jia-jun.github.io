``` C
#include <stdio.h>
int main(){
    int arr[5] = {1, 2, 3, 4, 5};

    // 以下兩種寫法相等
    // int *p = arr; // 取 arr 首地址存入 pointer p
    int *p = &arr[0]; // & 取地址，並賦值到 *p
    p++; // 使 p 指向下一個元素
    printf("%d\n", *p);
    return  0;
}
```

運算子結合是由右往左結合
``` C
#include <stdio.h>
int main(){
    int arr[5] = {1, 3, 5, 7, 9};

    // 以下兩種寫法相等
    // int *p = arr; // 取 arr 首地址存入 pointer p
    int *p = &arr[0]; // & 取地址，並賦值到 *p -> 1
    p++; // 使 p 指向下一個元素 -> 3
    printf("%d\n", *p); // 3
    printf("%d\n", *p++); //取值後輸出，並往後移動 4 個 bit -> 3
    printf("%d\n", *p); // 重新取用一次 p 的值 -> 5
    // *++p 運算子是由右往左結合，故等同於 *(++p)
    // ++p 為 7，*++p = 取 p 值 = 7
    printf("%d\n", *++p); // 再移動指針增加 4 bit, 並取值 p -> 7
    // ++*p = ++(*p) 取 *p 值並 +1(由右往左結合)
    printf("%d\n", ++*p); // 取指針 p 的值(7)，並將該值 + 1 -> 8

    return  0;
}
```
