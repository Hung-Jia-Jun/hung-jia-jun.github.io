```C
#include <string.h>
#include <stdio.h>

// 聲明一個結構體
struct Person
{
   char name[20];
   int age;
};

// 結構體指針
int main()
{
    struct Person per1 = {.name="Tom", .age=12};
    struct Person *per_ptr = &per1;

    // 兩者等價
    (*per_ptr).age = 20;
    per_ptr->age = 30;

    printf("name: %s\n", (*per_ptr).name);
    printf("age: %d\n", (*per_ptr).age);
    
}
```
