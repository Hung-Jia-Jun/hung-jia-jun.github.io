```C
#include <string.h>
#include <stdio.h>

struct Person
{
    char name[20];
    int age;
};

int main()
{
    // 1. 創建結構體數組
    struct Person p_arr[5];

    // 2. 如何給結構體數組的元素賦值    
    strcpy(p_arr[0].name, "Jason");
    p_arr[0].age = 5;
    strcpy(p_arr[1].name, "Jason111");
    p_arr[0].age = 6;
    strcpy(p_arr[2].name, "Jason222");
    p_arr[0].age = 7;
    //3. 如何調用數組的元素，即結構體內部的成員

    printf("p_arr[0].name: %s\n", p_arr[0].name);
    printf("p_arr[1].name: %s\n", p_arr[1].name);
    printf("p_arr[2].name: %s\n", p_arr[2].name);

    /*聲明方式 2*/
    // 1. 創建結構體數組
    struct Person p_arr2[]={ {.name="John", .age=1},
                            {.name="John1", .age=2},
                            {.name="Joh2", .age=3}};

    printf("p_arr2[0].name: %s\n", p_arr2[0].name);
    printf("p_arr2[1].name: %s\n", p_arr2[1].name);
    printf("p_arr2[2].name: %s\n", p_arr2[2].name);
    
    /*使用指向數組的指針*/
    struct Person *ptrPerson;
    
    // 取得第 0 個結構體的指針參考
    ptrPerson = &p_arr[0];
    printf("ptrPerson.name: %s\n", ptrPerson->name);
}
```
